Firmware:

Here is the code used for A4: Libraries. This code defines the buttons and then uses the buttons to light up different areas of the seven segment display via an IR remote library and Seven Segment Display library.


Note: Serial.println was used to print the values from the remote at first and then used to define the remote buttons


/*

  A3 Input Output

*/


void setup() {

  // initalizes pin 6 to fade yellow LED

  pinMode(6, OUTPUT);

  // initalizes pin 9 to fade blue LED

  pinMode(9, OUTPUT);

  // initalize serial communications at 9600 bps

  Serial.begin(9600);

}


void loop() {

  // initializes variable which is the value of the photoresistor:

  int val = analogRead(A0);

  // prints value of analogRead --> shows that its around 400 to 800

  Serial.println(val);

  // limits range of sensor values to between 400 and 800

  int sensVal = constrain(val, 400, 800);

  // connects photoresistor values to led brightness values 

  int outVal = map(sensVal, 400, 800, 0, 255);


  // if statement for when sensory value is above 600

  if (sensVal >= 600) {

    // makes yellow led brightness the photoresistor value

    analogWrite(6, outVal);

    // tuns off blue led

    digitalWrite(9, LOW);

  // else for when sensory value is below 600

  } else {

    // makes blue led brightness the photoresistor value

    analogWrite(9, outVal);

    // tuns off yellow led

    digitalWrite(6, LOW);

  }

}
