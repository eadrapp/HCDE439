Firmware:

Here is the code used for A2: Fade. This code initializes pins 9, 10, & 11 and then reads pin 2 and runs a fade loop based on the pin 2 reading.


/*

  A2 Fade

  Note: Thanks for Michelle Weng and David Song for the help

*/


void setup() {

  // initalizes pin 11 for red led

  pinMode(11, OUTPUT);

  // initalizes pin 10 for yellow led

  pinMode(10, OUTPUT);

  // initalizes pin 9 for green led

  pinMode(9, OUTPUT);

}


void loop() {

  // initializes variable which is the value of the button:

  int reading = digitalRead(2);

  // if the button is unpressed then the following if statement runs

  if (reading == 0){

    // for loop to fade leds brighter

    for (int fadeValue = 0; fadeValue <= 255; fadeValue += 5) {

      // increasing fade for red led

      analogWrite(11, fadeValue);

      // increasing fade for yellow led

      analogWrite(10, fadeValue);

      // increasing fade for green led

      analogWrite(9, fadeValue);

      // wait for 30 secs for dimming effect

      delay(30);

    }

    // for loop to fade leds dimmer

    for (int fadeValue = 255; fadeValue >= 0; fadeValue -= 5) {

      // decreasing fade for red led

      analogWrite(11, fadeValue);

      // decreasing fade for yellow led

      analogWrite(10, fadeValue);

      // decreasing fade for green led

      analogWrite(9, fadeValue);

      // wait for 30 secs for dimming effect

      delay(30);

    }

  // turns off led when button is pressed

  } else {

    // turns off red led

      analogWrite(11, 0);

    // turns off red yellow

      analogWrite(10, 0);

    // turns off red green

      analogWrite(9, 0);

  }

}
